Lista de Clases (Class List)

Que es classList

Es un atributo de los elementos HTML que contiene la DOMTokenList de clases del elemento. Se diferencia de className en que esta contiene solo texto plano.

id="minino" class="elementoDOM clase1 clase2 clase3"


let minino = document.getElementById('minino')

console.log(minino.classList); // DOMTokenList(4) ['elementoDOM', 'clase1', 'clase2', 'clase3', value: 'elementoDOM clase1 clase2 clase3']

console.log(minino.className); // elementoDOM clase1 clase2 clase3

Esto es lo que contiene en el caso de los inputs, es lo mismo, incluso con el atributoclassName:



let canino = document.getElementById('canino');

console.log(canino.className); // elementoDOM clase1 clase2 clase3

console.log(canino.classList); // DOMTokenList(4) ['elementoDOM', 'clase1', 'clase2', 'clase3', value: 'elementoDOM clase1 clase2 clase3']

Metodos de classList

Estos metodos sirven para manipular el atributo classList de los elementos.

add()

Este metodo agrega una nueva clase a la lista.

id="borrego"


let borrego = document.getElementById('borrego');

borrego.classList.add('txt-blue'); // El borrego se volvio azul!!

remove()

Este metodo remueve una clase de la lista.

id="ballena" class="elementoDOM txt-blue"


let ballena = document.getElementById('ballena');

ballena.classList.remove('txt-blue'); // La ballena se volvio blanca!!

Esto tambien se puede hacer con className... pero supongo que sera mejor y mas moderno trabajar las clases con la lista de clases.

let ballena = document.getElementById('ballena');

ballena.className = 'elementoDOM'; // La ballena se volvio blanca!!

item()

Devuelve la clase del indice especificado.

id="grandinosaurio" class="elementoDOM clase1 clase2 clase3"


let grandinosaurio = document.getElementById('grandinosaurio');

console.log(grandinosaurio.classList); // DOMTokenList(4) ['elementoDOM', 'clase1', 'clase2', 'clase3', value: 'elementoDOM clase1 clase2 clase3']

for (let i = 0; i < 4; i++) {
    console.log(
        grandinosaurio.classList.item(i)
    );
}

contains()

Verifica si el elemento posee o no la clase especificada.

id="raptor" class="elementoDOM clase1 clase2 clase3"


let raptor = document.getElementById('raptor');

let cont1 = raptor.classList.contains('clase1'),
    cont2 = raptor.classList.contains('latumpAbstrea');

console.log(cont1); // true
console.log(cont2); // false

replace()

Reemplaza una clase por otra.

id="paqui" class="elementoDOM clase1"


let paqui = document.getElementById('paqui');

paqui.classList.replace('clase1', 'txt-blue'); // El Paqui se puso azul!!

let newInfo = paqui.classList;

console.log(newInfo); // DOMTokenList(2) ['elementoDOM', 'txt-blue', value: 'elementoDOM txt-blue']

toggle()

Alterna una clase: si el elemento ya la tiene se la quita, si no la tiene se la pone.

id="ovejaMulticolor"


let ovejaMulticolor = document.getElementById('ovejaMulticolor'),
    btnToggleClass = document.getElementById('btnToggleClass');

btnToggleClass.addEventListener('click', funcion = () => {

    ovejaMulticolor.classList.toggle('txt-blue');

});
