Agregar Clave Externa

Al hacer referencia a los registros de otra tabla, la Clave Externa tambien es una Restriccion a la hora de eliminar algun registro incluido en otra tabla.

CREATE TABLE nombre_tabla (
    id int(11) NOT NULL AUTO_INCREMENT,
    clave_externa int(11) NOT NULL,
);

ALTER TABLE nombre_tabla
    ADD PRIMARY KEY (id),
    ADD KEY nombre_relacion_fk (clave_externa);

ALTER TABLE nombre_tabla
    ADD CONSTRAINT nombre_relacion_fk FOREIGN KEY (clave_externa) REFERENCES tabla_externa (id) ON DELETE reference_option ON UPDATE reference_option;


reference_option:
    RESTRICT | CASCADE | SET NULL | NO ACTION | SET DEFAULT

Al crear la tabla

CREATE TABLE nombre_tabla (
    id int(11) NOT NULL AUTO_INCREMENT,
    clave_externa int(11) NOT NULL,
    PRIMARY KEY (id)
    CONSTRAINT FOREIGN KEY (clave_externa) REFERENCES tabla_externa (id)
    ON DELETE reference_option ON UPDATE reference_option
);

reference_option:
    RESTRICT | CASCADE | SET NULL | NO ACTION | SET DEFAULT
