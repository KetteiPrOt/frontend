Decoradores

Son funciones especiales que decoran otras. Se usan para cosas como la extensión de una función sin modificarla. 


def decorator(function):
    def newFunction():
        print('Before execution')
        function()
        print('After execution')
    return newFunction

def greetings():
    print('Hello World!')

modifiedGreetings = decorator(greetings)
modifiedGreetings()
''''
Before execution
Hello World!
After execution
'''

La sintaxis correcta para realizar esto es la siguiente. Se obtiene el mismo resultado, con menos código: 


def decorator(function):
    def newFunction():
        print('Before execution')
        function()
        print('After execution')
    return newFunction

@decorator
def greetings():
    print('Hello World!')

greetings()
''''
Before execution
Hello World!
After execution
'''

@property

Este decorador es usado para invocar a los getters como si fueran atributos. 


class Person:
    __species = 'Homo Sapiens'

    def __init__(self, name):
        self.name = name

    @property
    def species(self):
        return self.__species

joel = Person('Joel')
print(joel.species) # Homo Sapiens

@property.setter

Este decorador es usado para invocar a los setters como si fueran atributos. 


class Person:
    __species = 'Homo Sapiens'

    def __init__(self, name):
        self.name = name

    @property
    def species(self):
        return self.__species
    
    @species.setter
    def species(self, species):
        self.__species = species

joel = Person('Joel')
joel.species = 'Alenígena'
print(joel.species) # Alienígena

@property.deleter

Un 'deleter' es un método que elimina el valor de una propiedad, como unset() en PHP, y con este decorador lo podemos llamar como si fuera un atributo. 


class Person:
    __species = 'Homo Sapiens'

    def __init__(self, name):
        self.name = name

    @property
    def species(self):
        return self.__species

    @species.deleter
    def species(self):
        del self.__species

joel = Person('Joel')
del joel.species # delete the property
print(joel.species) # AttributeError
